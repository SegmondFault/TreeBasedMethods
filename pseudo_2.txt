class Row
    features : list of numbers
    label : int   // 0 or 1


class Node
    feature_index : int
    threshold : number
    left : Node or Leaf
    right : Node or Leaf


class Leaf
    prediction : int       // 0 or 1
    probability : float    // probability of positive class
    pos_count : int
    neg_count : int


class Forest
    trees : list of Node   // each tree is stored by its root node


# --------------------------
# HELPER FUNCTIONS
# --------------------------

def purity(data_subset)
    if length(data_subset) == 0
        return true

    first_label = data_subset[0].label

    for row in data_subset
        if row.label != first_label
            return false

    return true


def count_labels(data_subset)
    pos = 0
    neg = 0

    for row in data_subset
        if row.label == 1
            pos += 1
        else
            neg += 1

    return pos, neg


def bootstrap_sample(data)
    N = length(data)
    sample = []

    for i in 0 to N - 1
        idx = random integer from 0 to N - 1
        append data[idx] to sample

    return sample

def split_data(data_subset, feature_index, threshold)
    left_data = []
    right_data = []

    for row in data_subset
        if row.features[feature_index] <= threshold
            append row to left_data
        else
            append row to right_data

    return left_data, right_data

def calc_split_score(left_data, right_data, criterion)
    total = length(left_data) + length(right_data)

    if total == 0
        return 0

    left_weight = length(left_data) / total
    right_weight = length(right_data) / total

    if criterion == "entropy"
        return left_weight * entropy(left_data) + right_weight * entropy(right_data)
    else
        return left_weight * gini(left_data) + right_weight * gini(right_data)

# --------------------------
# LEAF BUILDING
# --------------------------

def build_leaf(data_subset)
    pos, neg = count_labels(data_subset)

    if pos >= neg
        prediction = 1
    else
        prediction = 0

    total = pos + neg

    if total == 0
        probability = 0
    else
        probability = pos / total

    return Leaf(prediction, probability, pos, neg)


# --------------------------
# TREE BUILDING
# --------------------------

def build_tree(data_subset, depth, split_number, criterion, min_samples)

    if length(data_subset) == 0 or depth == 0 or purity(data_subset) == true or length(data_subset) < min_samples:
        return build_leaf(data_subset)

    total_feature_count = length(data_subset[0].features)
    features_subset = random_select_features(total_feature_count, split_number)

    best_feature = None
    best_threshold = None
    best_score = very_large_number

    for f in features_subset
        threshold, score = find_thresh(data_subset, f, criterion)

        if score < best_score
            best_score = score
            best_feature = f
            best_threshold = threshold

    if best_feature == None
        return build_leaf(data_subset)

    left_data, right_data = split_data(data_subset, best_feature, best_threshold)

    if length(left_data) == 0 or length(right_data) == 0
        return build_leaf(data_subset)

    left_node = build_tree(left_data, depth - 1, split_number, criterion, min_samples)
    right_node = build_tree(right_data, depth - 1, split_number, criterion, min_samples)

    return Node(best_feature, best_threshold, left_node, right_node)


# --------------------------
# TREE PREDICTION
# --------------------------

def run_tree(row, tree)
    current = tree

    while current is a Node
        if row.features[current.feature_index] <= current.threshold
            current = current.left
        else
            current = current.right

    return current.prediction


def run_tree_probability(row, tree)
    current = tree

    while current is a Node
        if row.features[current.feature_index] <= current.threshold
            current = current.left
        else
            current = current.right

    return current.probability


# --------------------------
# FOREST BUILDING
# --------------------------

def build_forest(data, num_trees, depth, split_number, criterion, min_samples)
    forest = Forest()
    forest.trees = []

    for i in 0 to num_trees - 1
        sample = bootstrap_sample(data)
        tree = build_tree(sample, depth, split_number, criterion, min_samples)
        append tree to forest.trees

    return forest


# --------------------------
# FOREST PREDICTION
# --------------------------

def majority_vote(votes)
    pos = 0
    neg = 0

    for v in votes
        if v == 1
            pos += 1
        else
            neg += 1

    if pos > neg
        return 1
    else
        return 0


def run_forest(row, forest)
    votes = []

    for tree in forest.trees
        prediction = run_tree(row, tree)
        append prediction to votes

    return majority_vote(votes)


def run_forest_probability(row, forest)
    probs = []

    for tree in forest.trees
        prob = run_tree_probability(row, tree)
        append prob to probs

    if length(probs) == 0
        return 0

    return average(probs)
